Algoritmo CAMBIAR_DOS_VALORES_NUMERICOS_USANDO_2_VARIABLES
	
	// CAMBIAR DOS VALORES NUMERICOS USANDO DOS VARIABLES:
	definir A, B Como Entero;
	
	NUM0 <- 5443 //ESTOS VALORES SON DE PRUEBA
	NUM1 <- 3435
	
	NUM1 <- NUM0+NUM1; 
	NUM0 <- NUM1-NUM0;  
	NUM1 <- NUM1-NUM0    
	Escribir "NUMERO1: ", NUM0, " NUMERO2: ", NUM1;
	
	// CAMBIAR DOS CARACTERES USANDO DOS VARIABLES:
	definir CHAR0, CHAR1 como caracter;
	CHAR0 <- "2"; //ESTOS VALORES SON DE PRUEBA
	CHAR1 <- "4";
	
	CHAR0 <- Concatenar(CHAR0, CHAR1);
	CHAR1 <- Subcadena(CHAR0, 1, 1);
	CHAR0 <- Subcadena(CHAR0, 2, 2);
	
	Escribir "CHAR0: ", CHAR0, " CHAR1: ", CHAR1;
	
	// CAMBIAR DOS CADENAS DE CARACTERES USANDO DOS VARIABLES:
	definir STRING0, STRING1 como texto;
	
	STRING0 <- "TU ERES UN CAGAGUATE SI" //ESTOS VALORES SON DE PRUEBA
	STRING1 <- "TU PERRA MADRE ES UNA PUTA"
	
	STRING0 <- Concatenar(STRING0, STRING1);
	Escribir "ESTE ES EL TEXTO CONCATENADO: ", STRING0, " Y SU LONGITUD ES: ", Longitud(STRING0);
	Escribir "ESTA ES LA LONGITUD DE CARACTERES QUE IRA A STRING1: ",(Longitud(STRING0)-longitud(STRING1));
	
	STRING1 <- Subcadena(STRING0,1, (Longitud(STRING0)-longitud(STRING1)))//ACA ACCEDEMOS AL INDICE 1 DE EL STRING0 Y RESTAMOS LA LONGITUD DE STRING0 - LA 
	                                                                     //LONGITUD DE STRING1 QUE DA COMO RESULTADO TODOS LOS CARACTERES DE STRING1
	
	Escribir "ESTA ES LA LONGITUD DE CARACTERES QUE IRA A STRING0: ",(Longitud(STRING0)-longitud(STRING1));
	STRING0 <- Subcadena(STRING0, Longitud(STRING1)+1, longitud(STRING0)) //ACA OBTENEMOS LA LONGITUD DE STRING1+1 PARA ACCEDER A LOS DEMAS ELEMENTOS DE STRING0
																	     // ES +1 PARA QUE EMPIEZE DESDE EL INDICE QUE QUEREMOS Y NO DESDE EL ULTIMO INDICE QUE
                                                                         // TIENE EL STRING0
	
	Escribir "STRING0: ", STRING0, " STRING1:", STRING1;
	
	//CAMBIAR DOS VALORES BOOLEANOS USANDO DOS VARIABLES:
	definir BOOL0, BOOL1 como logicos;
	
	BOOL0 <- falso; 
	BOOL1 <- Verdadero;
	
	BOOL0 <- (no BOOL0);
	
	BOOL1 <- (no BOOL1);
	
	Escribir "VALOR BOOLEANO0: ", BOOL0, " VALOR BOOLEANO1: ", BOOL1;
	
	//CAMBIAR DOS VALORES REALES USANDO DOS VARIABLES:
	Definir REAL0, REAL1 como reales;
	
	REAL0 <- 5.7564; //ESTOS VALORES SON DE PRUEBA
	REAL1 <- 9.7653;
	
	REAL0 <- REAL0+REAL1;
	REAL1 <- REAL0-REAL1;
	REAL0 <- REAL0-REAL1;
	
	Escribir "EL VALOR DE REAL0 ES: ", REAL0, " EL VALOR DE REAL1 ES: ", REAL1;
